# PR Review Dashboard — Plan 6: Frontend

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the complete React frontend: login page, PR list with All/Needs-My-Review tabs, metrics charts with date range selection, and a settings page for the repo allowlist.

**Architecture:** shadcn/ui + Tailwind for components and layout. TanStack Query for data fetching via the orval-generated typed hooks (from Plan 5). React Router for client-side routing. Charts use Recharts (verify Tremor license before substituting). The Vite dev server proxies `/api` to NestJS so auth cookies work during development.

**Tech Stack:** React 19, React Router v6, TanStack Query v5, Tailwind CSS, shadcn/ui, Recharts, orval-generated API client (from Plan 5)

---

## Design Decisions (from grill-me session)

- **Components:** shadcn/ui + Tailwind
- **Charts:** Recharts (Tremor as optional upgrade — verify MIT license at `github.com/tremorlabs/tremor` before using)
- **Data fetching:** TanStack Query + orval-generated hooks
- **Routing:** React Router (client-side, NestJS serves `index.html` as SPA fallback)
- **PR list tabs:** "All PRs" (all open) and "Needs My Review" (open PRs where current user hasn't approved)
- **Date range:** 7 / 30 / 90 day presets + custom date picker
- **Settings:** Repo allowlist (user picks which repos to show in their dashboard view)
- **Notifications:** None
- **Auth:** Login page redirects to `/api/auth/github`. After OAuth, GitHub redirects back. No in-app state management needed for auth tokens — cookies handled by the browser.

```tsx
function daysAgo(iso: string): string {
  const diff = Date.now() - new Date(iso).getTime();
```

```tsx
import { Button } from '../../components/ui/button';
import { cn } from '../../lib/utils';

export interface DateRange {
  from: string; // ISO date string YYYY-MM-DD
  to: string;
}

interface Props {
  value: DateRange;
  onChange: (range: DateRange) => void;
}

function preset(days: number): DateRange {
  const to = new Date();
  const from = new Date(Date.now() - days * 86_400_000);
  return {
    from: from.toISOString().slice(0, 10),
    to: to.toISOString().slice(0, 10),
  };
}

const PRESETS = [
  { label: '7 days', days: 7 },
  { label: '30 days', days: 30 },
  { label: '90 days', days: 90 },
];

export function DateRangePicker({ value, onChange }: Props) {
  return (
    <div className="flex items-center gap-2">
      {PRESETS.map((p) => {
        const range = preset(p.days);
        const active = value.from === range.from && value.to === range.to;
        return (
          <Button
            key={p.days}
            variant={active ? 'default' : 'outline'}
            size="sm"
            onClick={() => onChange(range)}
          >
            {p.label}
          </Button>
        );
      })}
      <input
        type="date"
        value={value.from}
        className="border rounded px-2 py-1 text-sm"
        onChange={(e) => onChange({ ...value, from: e.target.value })}
      />
      <span className="text-gray-400 text-sm">to</span>
      <input
        type="date"
        value={value.to}
        className="border rounded px-2 py-1 text-sm"
        onChange={(e) => onChange({ ...value, to: e.target.value })}
      />
    </div>
  );
}
```

- [ ] **Step 1: Create `apps/web/src/features/settings/RepoAllowlist.tsx`**

```tsx
import { useState } from 'react';
import { Button } from '../../components/ui/button';

interface Repo {
  id: number;
  fullName: string;
  isArchived: boolean;
}

interface Props {
  allRepos: Repo[];
  selectedIds: number[];
  onSave: (ids: number[]) => Promise<void>;
  isSaving: boolean;
}

export function RepoAllowlist({ allRepos, selectedIds, onSave, isSaving }: Props) {
  const [selected, setSelected] = useState<Set<number>>(new Set(selectedIds));

  function toggle(id: number) {
    setSelected((prev) => {
      const next = new Set(prev);
      next.has(id) ? next.delete(id) : next.add(id);
      return next;
    });
  }

  return (
    <div className="space-y-4">
      <p className="text-sm text-gray-500">
        Choose which repos appear in your PR list. All repos are synced regardless.
      </p>
      <div className="border rounded-lg divide-y max-h-96 overflow-y-auto">
        {allRepos.filter((r) => !r.isArchived).map((repo) => (
          <label key={repo.id} className="flex items-center gap-3 px-4 py-3 hover:bg-gray-50 cursor-pointer">
            <input
              type="checkbox"
              checked={selected.has(repo.id)}
              onChange={() => toggle(repo.id)}
              className="rounded"
            />
            <span className="text-sm font-medium">{repo.fullName}</span>
          </label>
        ))}
      </div>
      <Button onClick={() => onSave(Array.from(selected))} disabled={isSaving}>
        {isSaving ? 'Saving…' : 'Save allowlist'}
      </Button>
    </div>
  );
}
```
