# PR Review Dashboard — Plan 5: REST API

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement all NestJS REST endpoints with Swagger decorators, then generate a fully-typed TypeScript client for the React frontend using orval.

**Architecture:** Each domain (prs, metrics, users) gets its own NestJS module with a controller, service, and response DTOs. Controllers are decorated with `@nestjs/swagger` decorators so Swagger generates the OpenAPI spec. `orval` reads the spec from the running server and generates a typed fetch client + TanStack Query hooks directly into the web app. The `SessionGuard` from Plan 3 protects all routes.

**Tech Stack:** `@nestjs/swagger`, `class-transformer`, `class-validator`, orval (devDependency in web app), `@tanstack/react-query`

---

## Design Decisions (from grill-me session)

- **PR list fields:** title, number, repo (fullName), author (login + avatarUrl), reviewDecision, ciStatus, age (openedAt), commentCount, lastActivityAt, isDraft, state
- **PR list tabs:** All PRs (open) / Needs My Review (open PRs where the current user has not approved)
- **Metrics:** PR volume (opened/merged/closed) by date range and optionally by author; review + comment counts per org member; per-PR stats (time to merge, comment count, commitCount). Date range: 7/30/90-day presets + custom (ISO date params)
- **User routes:** GET /users/me (current session user), PATCH /users/me/allowlist (per-user repo allowlist)
- **SDK generation:** orval generates from `http://localhost:3000/api-json` into `apps/web/src/api/`
- **Auth:** All routes except `/health` require `SessionGuard`

```typescript
import { Injectable } from '@nestjs/common';
import { PrState } from '@prisma/client';
import { plainToInstance } from 'class-transformer';
import { PrismaService } from '../prisma/prisma.service';
import { PrListItemDto } from './dto/pr-list-item.dto';
import { PrQueryDto, PrTab } from './dto/pr-query.dto';

const PAGE_SIZE = 50;

@Injectable()
export class PrsService {
  constructor(private readonly prisma: PrismaService) {}

  async findAll(query: PrQueryDto): Promise<{ items: PrListItemDto[]; total: number; page: number }> {
    const page = query.page ?? 1;
    const where = {
      state: PrState.OPEN,
      ...(query.authorLogin ? { authorLogin: query.authorLogin } : {}),
      ...(query.repoId ? { repoId: query.repoId } : {}),
    };

    const [raw, total] = await Promise.all([
      this.prisma.pullRequest.findMany({
        where,
        include: { repo: { select: { fullName: true } } },
        orderBy: { lastActivityAt: 'desc' },
        skip: (page - 1) * PAGE_SIZE,
        take: PAGE_SIZE,
      }),
      this.prisma.pullRequest.count({ where }),
    ]);

    const items = raw.map((pr) =>
      plainToInstance(PrListItemDto, { ...pr, repoFullName: pr.repo.fullName }, { excludeExtraneousValues: true }),
    );

    return { items, total, page };
  }
}
```

```typescript
import { Injectable } from '@nestjs/common';
import { PrismaService } from '../prisma/prisma.service';
import { MetricsQueryDto } from './dto/metrics-query.dto';
import { ReviewerStatsDto, VolumePointDto } from './dto/volume-point.dto';

@Injectable()
export class MetricsService {
  constructor(private readonly prisma: PrismaService) {}

  async getVolume(query: MetricsQueryDto): Promise<VolumePointDto[]> {
    const from = new Date(query.from);
    const to = new Date(query.to);
    to.setHours(23, 59, 59, 999);

    const prs = await this.prisma.pullRequest.findMany({
      where: {
        openedAt: { gte: from, lte: to },
        ...(query.authorLogin ? { authorLogin: query.authorLogin } : {}),
      },
      select: { openedAt: true, mergedAt: true, closedAt: true },
    });

    const byDay = new Map<string, VolumePointDto>();

    const dayKey = (d: Date) => d.toISOString().slice(0, 10);

    // Fill every day in the range
    for (let d = new Date(from); d <= to; d.setDate(d.getDate() + 1)) {
      const key = dayKey(new Date(d));
      byDay.set(key, { date: key, opened: 0, merged: 0, closed: 0 });
    }

    for (const pr of prs) {
      const openKey = dayKey(pr.openedAt);
      if (byDay.has(openKey)) byDay.get(openKey)!.opened++;
      if (pr.mergedAt) {
        const mergeKey = dayKey(pr.mergedAt);
        if (byDay.has(mergeKey)) byDay.get(mergeKey)!.merged++;
      } else if (pr.closedAt) {
        const closeKey = dayKey(pr.closedAt);
        if (byDay.has(closeKey)) byDay.get(closeKey)!.closed++;
      }
    }

    return Array.from(byDay.values()).sort((a, b) => a.date.localeCompare(b.date));
  }

  async getReviewerStats(query: MetricsQueryDto): Promise<ReviewerStatsDto[]> {
    const from = new Date(query.from);
    const to = new Date(query.to);
    to.setHours(23, 59, 59, 999);

    const reviews = await this.prisma.review.findMany({
      where: { submittedAt: { gte: from, lte: to } },
      select: { reviewerLogin: true },
    });

    const counts = new Map<string, number>();
    for (const r of reviews) {
      counts.set(r.reviewerLogin, (counts.get(r.reviewerLogin) ?? 0) + 1);
    }

    return Array.from(counts.entries())
      .map(([login, reviewCount]) => ({ login, reviewCount, commentCount: 0 }))
      .sort((a, b) => b.reviewCount - a.reviewCount);
  }
}
```

```typescript
import { Body, Controller, Get, Put, Req, UseGuards } from '@nestjs/common';
import { ApiOkResponse, ApiTags } from '@nestjs/swagger';
import { Request } from 'express';
import { User } from '@prisma/client';
import { SessionGuard } from '../auth/session.guard';
import { UpdateAllowlistDto } from './dto/update-allowlist.dto';
import { UsersService } from './users.service';

@ApiTags('users')
@UseGuards(SessionGuard)
@Controller('users')
export class UsersController {
  constructor(private readonly users: UsersService) {}

  @Get('me')
  @ApiOkResponse({ description: 'Current user' })
  me(@Req() req: Request): User {
    return req.user as User;
  }

  @Put('me/allowlist')
  @ApiOkResponse({ description: 'Repo allowlist replaced' })
  async setAllowlist(@Req() req: Request, @Body() dto: UpdateAllowlistDto): Promise<void> {
    const user = req.user as User;
    await this.users.setAllowlist(user.id, dto.repoIds);
  }
}
```

```typescript
export const customFetch = async <T>(url: string, options: RequestInit): Promise<T> => {
  const response = await fetch(url, { ...options, credentials: 'include' });
  if (!response.ok) {
    throw new Error(`${response.status} ${response.statusText}`);
  }
  return response.json() as Promise<T>;
```
